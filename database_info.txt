Copy paste for easy creation of tables in db

CREATE TABLE Groups (
    group_id SERIAL PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    code VARCHAR(10) NOT NULL,
    created_on DATE NOT NULL
);

CREATE TABLE Users (
  chat_id INT PRIMARY KEY,
  group_id INT NOT NULL,
  name VARCHAR(50) NOT NULL,
  dob DATE,
  code VARCHAR(10) UNIQUE NOT NULL,
  joined_on DATE NOT NULL,

  FOREIGN KEY (group_id) REFERENCES Groups (group_id) ON DELETE CASCADE ON UPDATE CASCADE,
);

CREATE TABLE Messages (
    message_id SERIAL PRIMARY KEY,
    user_code_from VARCHAR(10) NOT NULL,
    user_code_to VARCHAR(10) NOT NULL,
    message VARCHAR(255) NOT NULL,
    message_sent BOOL NOT NULL,

    FOREIGN KEY (user_code_from) REFERENCES Users (code) ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (user_code_to) REFERENCES Users (code) ON DELETE CASCADE ON UPDATE CASCADE
);

To "reset" database (clear all rows)
delete from Groups;
delete from Users;
